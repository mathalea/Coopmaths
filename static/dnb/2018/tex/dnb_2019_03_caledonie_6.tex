\begin{scratch}
\blockinit{quand \greenflag est cliqué}
\blockpen{effacer tout}
\blockmove{s’orienter à \ovalnum{90\selectarrownum}}
\blocksensing{demander \txtbox{Donne une longueur} et attendre}
\blockvariable{mettre \selectmenu{longueur} à \ovalsensing{réponse}}
\blocksensing{demander \txtbox{Donne un angle} et attendre}
\blockvariable{mettre \selectmenu{angle} à \ovalsensing{réponse}}
\blockpen{stylo en position d’écriture}
\blockmove{avancer de \ovalnum{100}}
\blockmove{tourner \turnright{} de \ovalvariable{angle} degrés}
\blockmove{avancer de \ovalvariable{longueur}}
\blockmove{tourner \turnright{} de \ovaloperator{\ovalnum{180} - \ovalvariable{angle}} degrés}
\blockmove{avancer de \ovalnum{100}}
\blockmove{tourner \turnright{} de \ovalvariable{angle} degrés}
\blockmove{avancer de \ovalvariable{longueur}}
\blockmove{tourner \turnright{} de \ovaloperator{\ovalnum{180} - \ovalvariable{angle}} degrés}
\blockpen{relever le stylo}
\end{scratch}}
