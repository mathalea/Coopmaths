\begin{enumerate}
\item Il y a 8 assemblages possibles.
\item  $p(\text{montre toute rouge}) = \dfrac{1}{8}$.
\item  $p(\text{montre d'une seule couleur}) = \dfrac{2}{8} = \dfrac{1}{4}$.
\item  $p(\text{montre de deux couleurs}) = \dfrac{6}{8} = \dfrac{3}{4}$.
\end{enumerate}
